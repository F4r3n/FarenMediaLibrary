foreach(TARGET freetype2 SDL2 ECS Box2D lua assimp openal glew)

	if(${TARGET} STREQUAL "lua")
	set(BUILD_SHARED_LIBS True)
	endif()

	if(${TARGET} STREQUAL "assimp")
	set(BUILD_SHARED_LIBS True)
	endif()
	add_subdirectory(${TARGET})
    set(BUILD_SHARED_LIBS OFF)
endforeach()
