
#Box2D
set(BOX2D_BUILD_STATIC OFF CACHE BOOL "" FORCE)
set(BOX2D_BUILD_SHARED ON CACHE BOOL "" FORCE)
set(BOX2D_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(BOX2D_INSTALL_BY_DEFAULT OFF CACHE BOOL "" FORCE)
set(BOX2D_INSTALL OFF CACHE BOOL "" FORCE)


#ECS
set(ECS_BUILD_STATIC OFF CACHE BOOL "" FORCE)
set(ECS_BUILD_SHARED ON CACHE BOOL "" FORCE)
set(ECS_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(ECS_BUILD_TESTS OFF CACHE BOOL "" FORCE)

foreach(TARGET freetype2 SDL2 ECS glew Box2D lua assimp)
set (CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/${TARGET}/lib)
set (CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/${TARGET}/lib)

	if(${TARGET} STREQUAL "freetype2")
	set(BUILD_SHARED_LIBS True)
	endif()

	if(${TARGET} STREQUAL "lua")
	set(BUILD_SHARED_LIBS True)
	endif()
	add_subdirectory(${TARGET})
	set(BUILD_SHARED_LIBS FALSE)
endforeach()
